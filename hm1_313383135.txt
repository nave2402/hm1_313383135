        x3 = float(x3)

def convert(hours=0, minutes=0):
    try:
        hours = float(hours)
        minutes = float(minutes)
        if hours >= 0 and minutes >= 0:
            hours_min = hours * 60
            total_min = hours_min + minutes
            seconds = total_min * 60
        return f"It will be {int(seconds)} seconds"
    except:
        return "Input error!"
